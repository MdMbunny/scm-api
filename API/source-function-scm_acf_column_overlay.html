 69: /**

 71: *
 72: * @param {string} name Required. Field Group name.
 73: * @param {string} key Required. Unique key.
 74: * @param {array=} attr Optional. List of attributes (default is empty array).
 75: * @return {array} Field Group settings.
 76: */
 77: function scm_acf_group( $name, $key, $attr = array() ) {
 78: 
 79:     if( !$name || !$key ) return;
 80: 
 81:     $group = array (
 82:         'key' => $key,
 83:         'title' => $name,
 84:         'fields' => array(),
 85:         'location' => array(
 86:             array(
 87:                 scm_acf_group_location_rule( 'post' ),
 88:                 scm_acf_group_location_rule_not( 'post' ),
 89:             ),
 90:         ),
 91:         'menu_order' => 0,
 92:         'position' => 'normal',
 93:         'style' => 'default',
 94:         'label_placement' => 'top',
 95:         'instruction_placement' => 'label',
 96:         'hide_on_screen' => array(
 97:             0 => 'the_content',
 98:             1 => 'excerpt',
 99:             2 => 'custom_fields',
100:             3 => 'discussion',
101:             4 => 'comments',
102:             5 => 'revisions',
103:             6 => 'slug',
104:             7 => 'author',
105:             8 => 'format',
106:             9 => 'page_attributes',
107:             10 => 'featured_image',
108:             11 => 'categories',
109:             12 => 'tags',
110:             13 => 'send-trackbacks',
111:         ),
112:     );
113: 
114:     if( is_array( $attr ) )
115:         $group = array_merge( $group, $attr );
116: 
117:     return $group;
118: }

120: /**
121: * [GET] Field Group location group
122: *
123: * @param {string|array=} value Single location or array of locations (default is empty array).
124: * @param {string=} param Location parameter (default is 'post_type').
125: * @param {string=} cond Location condition (default is '==').
126: * @return {array} Location group.
127: */
128: function scm_acf_group_location( $value = array(), $param = 'post_type', $cond = '==' ) {
129: 
130:     $location = array();
131:     $value = toArray( $value, false, true );
132:     if( !$value || is_asso( $value ) ) return $location;
133: 
134:     foreach( $value as $loc )
135:         $location[] = scm_acf_group_location_rule( $loc, $param, $cond );
136: 
137:     return $location;
138: }

140: /**
141: * [GET] Field Group location rule
142: *
143: * @param {string} value Required. Single location.
144: * @param {string=} param Location parameter (default is 'post_type').
145: * @param {string=} cond Location condition (default is '==').
146: * @return {array} Location rule.
147: */
148: function scm_acf_group_location_rule( $value = NULL, $param = 'post_type', $cond = '==' ) {
149:     $rule = array();
150:     if( is_null( $value ) || !isset( $value ) || !$value || !is_string( $value ) ) return $rule;
151:     
152:     $value = ( startsWith( $value, 'admin' ) ? 'administrator' : $value );
153:     $cond = ( $cond ?: '==' );
154:     $param = ( $param ?: ( $value == 'administrator' ? 'current_user_role' : 'post_type') );
155: 
156:     $rule = array (
157:         'param' => $param,
158:         'operator' => $cond,
159:         'value' => $value,
160:     );
161: 
162:     return $rule;
163: }

179: /**
180: * [GET] Field Group set keys to fields and return
181: *
182: * @param {string} key Key base.
183: * @param {array} list List of fields.
184: * @return {array} Modified list of fields.
185: */
186: function scm_acf_group_keys( $key, $list ) {
187: 
188:     if( !$key || !$list )
189:         return array();
190: 
191:     for ( $i = 0; $i < sizeof( $list ); $i++ ) {
192: 
193:         $new = $key . ( substr( $key, -1 ) != '_' ? '_' : '' ) . $list[$i]['key'];
194:         $list[$i]['key'] = $new;
195: 
196:         if( isset( $list[$i]['layouts'] ) ){
197:             
198:             for ( $j = 0; $j < sizeof( $list[$i]['layouts'] ); $j++ ) {
199:                 $lay = $list[$i]['layouts'][$j]['key'] = 'layout_' . hash('ripemd160', $new . $list[$i]['layouts'][$j]['name'] );
200:                 $list[$i]['layouts'][$j]['sub_fields'] = scm_acf_group_keys( $lay, $list[$i]['layouts'][$j]['sub_fields'] );
201:             }
202: 
203:         }else if( isset( $list[$i]['sub_fields'] ) ){
204:             $list[$i]['sub_fields'] = scm_acf_group_keys( $new, $list[$i]['sub_fields'] );
205:         }
206: 
207:         $list[$i]['key'] = 'field_' . hash('ripemd160', $list[$i]['key'] . $list[$i]['type'] );
208:         $list[$i] = scm_acf_group_conditional( $list[$i], $list );
209:     }
210: 
211:     return $list;
212: }

214: /**
215: * [GET] Field Group conditional logics merging
216: *
217: * @param {array} ... One or more conditional logic array.
218: * @return {array} Conditional logic array.
219: */
220: function scm_acf_group_condition() {
221: 
222:     // Get arguments
223:     if( !func_num_args() ) return;
224:     $logic = func_get_args();
225: 
226:     $arr = array();
227: 
228:     foreach ($logic as $value) {
229:         if( !$value || !is_array( $value ) ) continue;
230:         if( !isset( $value[0] ) ){
231:             $arr[] = $value;
232:         }else{
233:             foreach ($value as $sub) {
234:                 if( !$sub || !is_array( $sub ) )
235:                     continue;
236:                 if( !isset( $sub[0] ) ){
237:                     $arr[] = $sub;              
238:                 }else{
239:                     foreach ($sub as $last) {
240:                         if( !$last || !is_array( $last ) )
241:                             continue;
242:                         if( !isset( $last[0] ) )
243:                             $arr[] = $last;
244:                     }
245:                 }
246:             }
247:         }
248:     }
249: 
250:     return $arr;
251: }
252: 
253: /**
254: * [GET] Field Group conditional logic
255: *
256: * @param {array} group Field Group.
257: * @param {array} list List of fields.
258: * @return {array} Modified Field Group.
259: */
260: function scm_acf_group_conditional( $group, $list ) {
261: 
262:     if( !$group || !$list )
263:         return $group;
264: 
265:     if( !is( $group['conditional_logic'] ) )
266:         return $group;
267: 
268:     $cond = $group['conditional_logic'];
269: 
270:     for ( $i = 0; $i < sizeof( $cond ); $i++ ) {
271:         for ( $j = 0; $j < sizeof( $cond[$i] ); $j++ ) {
272:             $name = $cond[$i][$j]['field'];
273:             $field = getByValueKey( $list, $name );
274:             if( !is_null( $field ) )
275:                 $cond[$i][$j]['field'] = $list[$field]['key'];
276:         }
277:     }
278: 
279:     $group['conditional_logic'] = $cond;
280: 
281:     return $group;
282: }
