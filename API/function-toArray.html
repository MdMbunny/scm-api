Function toArray

[GET] Convert item to array

Example usage:

// Anything to array
$var = 'variable';
$new = toArray( $var );
print( $new ) // [ 'variable' ]

// Empty to array
$var = '';
$new = toArray( $var );
print( $new ) // ['']

// Empty to array ($empty=true)
$emp = '';
$new = toArray( $emp, false, true );
print( $new ) // ''

// Indexed array to array
$array = [ 'value' ];
$new = toArray( $array );
print( $new ) // [ 'value' ]

// Associative array to array
$asso = [ 'key' => 'value' ];
$new = toArray( $asso );
print( $new ) // [ 'key' => 'value' ]

// Associative array to array ($asso=true)
$asso = [ 'key' => 'value' ];
$new = toArray( $asso, true );
print( $new ) // [ [ 'key' => 'value' ] ]
